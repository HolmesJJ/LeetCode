### 题目
https://leetcode.com/problems/sudoku-solver/

### 分类
* 回溯

### 思路1
* 注意边界条件

### 答案
```
public void solveSudoku(char[][] board) {
    backtracking(board, 0, 0);
}

public boolean backtracking(char[][] board, int m, int n) {
    
    // 是否已经完成
    boolean isDone = false;
    
    int a = board.length;
    int b = board[0].length;
    
    // 找到下一个点的位置
    int nm = m;
    int nn = n;
    while (nm < a && nn < b && board[nm][nn] != '.') {
        // 下一个结点，从左到右，再从上到下
        if (nn + 1 >= b) {
            nm = nm + 1;
            nn = 0;
        } else {
            nn = nn + 1;
        }
    }

    // 收集结果，代表board已经填满，当前结果为唯一解
    if (nm >= a) {
        isDone = true;
        return isDone;
    }
    
    // 确定当前该点的位置的行，列和3x3格子已经使用的值
    Map<Character, Character> map = new HashMap<>();
    // 行
    for (int i = 0; i < a; i++) {
        if (board[nm][i] != '.') {
            map.put(board[nm][i], board[nm][i]);
        }
    }
    // 列
    for (int i = 0; i < b; i++) {
        if (board[i][nn] != '.') {
            map.put(board[i][nn], board[i][nn]);
        }
    }
    // 3x3格子
    for (int i = nm / 3 * 3; i <= nm / 3 * 3 + 2; i++) {
        for (int j = nn / 3 * 3; j <= nn / 3 * 3 + 2; j++) {
            if (board[i][j] != '.') {
                map.put(board[i][j], board[i][j]);
            }
        }
    }
    
    // 遍历集合元素
    // 可以填入1 - 9个数字
    for (int i = 1; i <= 9; i++) {
        char num = (char) (i + '0');
        // 处理结点
        if (map.containsKey(num)) {
            continue;
        }
        // 只执行没有使用的值
        board[nm][nn] = num;
        isDone = backtracking(board, nm, nn);
        if (isDone) {
            break;
        }
        // 回溯操作
        board[nm][nn] = '.';
    }
    return isDone;
}
```

### 复杂度
* 时间复杂度：O((9!)^9)
* 空间复杂度：O(1)
